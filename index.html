function toWeirdCase(string){
  let arr = string.split(' ').map((element) => {
    let res = ''
    for (let i = 0; i < element.length; i++) {
      if (i % 2 === 0) {
        res += element[i].toUpperCase()
      } else {
        res += element[i]
      }
    }
    return res
  })
  return arr.join(' ')
}
